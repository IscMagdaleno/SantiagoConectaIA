ALTER PROCEDURE spGetChat (
@iIdProyecto INT 
) 
AS 
BEGIN 


CREATE TABLE #Result (
	bResult BIT DEFAULT (1),
	vchMessage VARCHAR(500) DEFAULT(''),
	 iIdChat INT DEFAULT( -1 ),
	 iIdProyecto INT DEFAULT( -1 ),
	 dtFechaCreacion DATETIME DEFAULT( '1900-01-01' ),
	 nvchNombre NVARCHAR (100)  DEFAULT( '' ),
	 bActivo BIT DEFAULT( 0 ),
   nvchThreadId NVARCHAR (100)  DEFAULT( '' ),
);

SET NOCOUNT ON

	BEGIN TRY

	INSERT INTO  #Result
	 ( 

		iIdChat, 			iIdProyecto, 			dtFechaCreacion, 	
		nvchNombre, 			bActivo 	,nvchThreadId		)
		SELECT 
		 C.iIdChat, 			 C.iIdProyecto, 			 C.dtFechaCreacion, 	
				 C.nvchNombre, 			 C.bActivo, nvchThreadId
         FROM
		 dbo.Chat C  WITH(NOLOCK)  
WHERE C.iIdProyecto =@iIdProyecto

		IF NOT EXISTS (SELECT 1 FROM #Result)
			BEGIN
				INSERT INTO #Result (bResult, vchMessage)
				SELECT 0, 'No register found.';
			END
	END TRY
	BEGIN CATCH
		INSERT INTO #Result (bResult, vchMessage)
		SELECT 0, CONCAT(ERROR_PROCEDURE(), ' : ', ERROR_MESSAGE(), ' - ', ERROR_LINE());
		PRINT CONCAT(ERROR_PROCEDURE(), ' : ', ERROR_MESSAGE(), ' - ', ERROR_LINE());
	END CATCH
	SELECT * FROM #Result;
	DROP TABLE #Result;
	END
